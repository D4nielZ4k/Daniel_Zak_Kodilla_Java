SELECT I.ID AS ID_ISSUES, I.SUMMARY, I.DESCRIPTION, I.USER_ID_ASSIGNEDTO AS USER_ID, Q.NAME
FROM ISSUES I JOIN ISSUESLISTS Q ON I.ISSUESLIST_ID = Q.ID;

SELECT I.ID AS ID_ISSUES, I.SUMMARY, I.DESCRIPTION, I.USER_ID_ASSIGNEDTO AS USER_ID, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I  JOIN USERS U ON U.ID = I.USER_ID_ASSIGNEDTO ;


SELECT USERS.FIRSTNAME, USERS.LASTNAME, COUNT(ISSUES.ID) AS ISSUES_NUMBER
FROM ISSUES INNER JOIN USERS ON USERS.ID=ISSUES.USER_ID_ASSIGNEDTO GROUP BY USERS.ID;
